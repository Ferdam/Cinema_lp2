CREATE SCHEMA CINEMA_DB;

-- DIRETOR
CREATE TABLE cinema_db.diretor (
id_diretor INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
nome VARCHAR(40) NOT NULL
);
-- GENERO
CREATE TABLE cinema_db.genero (
id_genero INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
nome VARCHAR(40) NOT NULL
);
-- ATOR
CREATE TABLE cinema_db.ator ( 
id_ator INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
nome VARCHAR(40) NOT NULL
);
